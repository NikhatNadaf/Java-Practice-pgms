package strings;


public class Strings {

    
    public static void main(String[] args) 
    {
        String str="123";
        System.out.println(str+10); //str concate with 10 as 12310
        int num=Integer.parseInt(str);  //converting string into number
        System.out.println(num+10); //becoz it number so it add in 10;
          
      //parseInt() from Integer class taking string as input & return integer as output
    }
    
}
